> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CMake "meta-project" that demonstrates deep knowledge of the C++ build and tooling ecosystem — sanitizers, LTO/PGO, cross-compilation, static analysis, compiler intrinsics, and assembly output — all driven by `CMakePresets.json`.

**Architecture:** Single CMake project with multiple demo executables. Shared CMake modules from `cmake/modules/` are imported.

**Tech Stack:** CMake 3.25+, GCC/Clang, GoogleTest, lcov/gcovr, clang-tidy, cppcheck, ARM cross-compiler (optional)

```cpp
#include <cstdlib>
#include <iostream>
#include <vector>

// MODE env var selects which bug to trigger:
//   clean       - no bug (default)
//   uaf         - heap-use-after-free
//   leak        - memory leak
//   oob         - out-of-bounds read

int main(int argc, char* argv[]) {
    const char* mode = (argc > 1) ? argv[1] : "clean";
    std::string m{mode};

    if (m == "uaf") {
        // Heap-use-after-free: ASan should catch this
        int* p = new int(42);
        delete p;
        std::cout << *p << "\n";  // USE AFTER FREE
    } else if (m == "leak") {
        // Memory leak: LSan (part of ASan) catches this
        int* p = new int[1024];
        (void)p;
        // intentionally not deleted
    } else if (m == "oob") {
        // Out-of-bounds: ASan should catch this
        std::vector<int> v{1, 2, 3};
        std::cout << v[10] << "\n";  // OOB
    } else {
        std::cout << "ASan demo running clean — no bug triggered.\n";
        std::cout << "Run with: uaf | leak | oob to trigger bugs.\n";
    }

    return 0;
}
```

```cpp
#include <thread>

// MODE: clean | race
// TSan should detect the data race in "race" mode.

int shared_counter = 0;  // intentionally not atomic

void increment(int n) {
    for (int i = 0; i < n; ++i)
        ++shared_counter;  // DATA RACE: two threads write without synchronization
}

int main(int argc, char* argv[]) {
    const char* mode = (argc > 1) ? argv[1] : "clean";
    std::string m{mode};

    if (m == "race") {
        shared_counter = 0;
        std::thread t1(increment, 100000);
        std::thread t2(increment, 100000);
        t1.join();
        t2.join();
        std::cout << "Final counter: " << shared_counter
                  << " (should be 200000 but won't be)\n";
    } else {
        std::cout << "TSan demo running clean.\n";
        std::cout << "Run with: race  to trigger a data race.\n";
```

```cpp
    const char* mode = (argc > 1) ? argv[1] : "clean";
```

Create `/home/zaki/workspaces/cpp/projects/01-toolchain/src/pgo_workload.cpp`:

```cpp
#include <algorithm>
#include <chrono>
#include <iostream>
#include <numeric>
#include <random>
#include <vector>

// A compute-heavy workload representative enough for PGO to find hot paths.
// Heavy branch: quicksort-like partition (95% of calls go the fast path).

static long bubble_sum(std::vector<int>& v) {
    // Intentionally naive: PGO won't optimize what it doesn't see run.
    long s = 0;
    for (auto x : v) s += x;
    return s;
}

static void sort_chunks(std::vector<int>& v, int chunk_size) {
    for (size_t i = 0; i + chunk_size <= v.size(); i += chunk_size)
        std::sort(v.begin() + i, v.begin() + i + chunk_size);
}

int main() {
    constexpr int N = 1'000'000;
    constexpr int ITERATIONS = 20;

    std::mt19937 rng{42};
    std::uniform_int_distribution<int> dist{0, 10'000};

    using clock = std::chrono::high_resolution_clock;
    long total_sum = 0;
    long total_ns = 0;

    for (int iter = 0; iter < ITERATIONS; ++iter) {
        std::vector<int> v(N);
        std::generate(v.begin(), v.end(), [&]{ return dist(rng); });

        auto t0 = clock::now();
        sort_chunks(v, 1000);
        total_sum += bubble_sum(v);
        auto t1 = clock::now();

        total_ns += std::chrono::duration_cast<std::chrono::nanoseconds>(t1 - t0).count();
    }

    std::cout << "Sum: " << total_sum << "\n";
    std::cout << "Avg time per iteration: " << (total_ns / ITERATIONS / 1'000) << " µs\n";
    return 0;
}
```

Create `/home/zaki/workspaces/cpp/projects/01-toolchain/src/intrinsics_demo.cpp`:

```cpp
#include <chrono>
#include <cstdint>
#include <cstring>
#include <immintrin.h>  // SSE/AVX intrinsics
#include <iostream>
#include <vector>

// CPUID to check at runtime whether AVX2 is available.
static bool cpu_has_avx2() {
#if defined(__x86_64__) || defined(__i386__)
    uint32_t eax, ebx, ecx, edx;
    // CPUID leaf 7, subleaf 0 → EBX bit 5 = AVX2
    __asm__ volatile(
        "cpuid"
        : "=a"(eax), "=b"(ebx), "=c"(ecx), "=d"(edx)
        : "a"(7), "c"(0)
    );
    return (ebx & (1u << 5)) != 0;
#else
    return false;
#endif
}

// Scalar dot product — baseline
static float dot_scalar(const float* a, const float* b, int n) {
    float sum = 0.0f;
    for (int i = 0; i < n; ++i) sum += a[i] * b[i];
    return sum;
}

// AVX2 dot product — 8 floats per iteration
static float dot_avx2(const float* a, const float* b, int n) {
    __m256 acc = _mm256_setzero_ps();
    int i = 0;
    for (; i + 8 <= n; i += 8) {
        __m256 va = _mm256_loadu_ps(a + i);
        __m256 vb = _mm256_loadu_ps(b + i);
        acc = _mm256_fmadd_ps(va, vb, acc);  // FMA: acc += va * vb
    }
    // Horizontal sum
    __m128 hi = _mm256_extractf128_ps(acc, 1);
    __m128 lo = _mm256_castps256_ps128(acc);
    lo = _mm_add_ps(lo, hi);
    lo = _mm_hadd_ps(lo, lo);
    lo = _mm_hadd_ps(lo, lo);
    float result = _mm_cvtss_f32(lo);
    // Scalar tail
    for (; i < n; ++i) result += a[i] * b[i];
    return result;
}

int main() {
    constexpr int N = 1 << 20;  // 1M elements
    std::vector<float> a(N, 1.0f), b(N, 2.0f);

    using clock = std::chrono::high_resolution_clock;

    std::cout << "CPU has AVX2: " << (cpu_has_avx2() ? "yes" : "no") << "\n\n";

    // Scalar
    auto t0 = clock::now();
    float s = dot_scalar(a.data(), b.data(), N);
    auto t1 = clock::now();
    auto scalar_us = std::chrono::duration_cast<std::chrono::microseconds>(t1 - t0).count();
    std::cout << "Scalar:  result=" << s << "  time=" << scalar_us << " µs\n";

    if (cpu_has_avx2()) {
        auto t2 = clock::now();
        float v = dot_avx2(a.data(), b.data(), N);
        auto t3 = clock::now();
        auto avx2_us = std::chrono::duration_cast<std::chrono::microseconds>(t3 - t2).count();
        std::cout << "AVX2:    result=" << v << "  time=" << avx2_us << " µs\n";
        std::cout << "Speedup: " << (float)scalar_us / avx2_us << "x\n";
    }
    return 0;
}
```
